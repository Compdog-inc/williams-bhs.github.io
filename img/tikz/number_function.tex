            \coordinate (outend) at ($(block.east)+(1.5,0)$);

            \draw[->] (block.east) -- (outend);

            \node at ($(instart)+(-0.2,0)$) {\large $x$};
            \node at ($(outend)+(0.2,0)$) {\large $y$};
